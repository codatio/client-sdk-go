)

func main() {
    s := codatbankfeeds.New(
        codatbankfeeds.WithSecurity(shared.Security{
            AuthHeader: "Basic BASE_64_ENCODED(API_KEY)",

    res, err := s.Companies.Create(ctx, shared.CompanyRequestBody{

}
